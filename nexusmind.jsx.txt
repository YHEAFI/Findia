const FALLBACK_IMG = "https://placehold.co/400x300/1a1a2e/e94560?text=No+Image";
const MODEL = "claude-sonnet-4-20250514";

// ─── AI FETCH ─────────────────────────────────────────────────────────────────
async function fetchProductsFromAI(query) {
  const prompt = `You are a real-time product search engine. The user is searching for: "${query}"

Return ONLY a valid JSON array (no markdown, no explanation) with exactly 6 product results.

Use your knowledge of real products, real pricing, and real retailers. Be accurate.

STRICT JSON format — every field required:
[
  {
    "name": "Full product name with model/variant",
    "price": "Price with currency symbol e.g. $299.99 or ₹24,999",
    "image": "https://... (a real plausible product image URL from the web, or null)",
    "source": "Retailer name e.g. Amazon, Walmart, Flipkart, Best Buy",
    "description": "2-sentence description of key features and why it's worth buying",
    "link": "https://www.amazon.com/s?k=${encodeURIComponent(query)} (use real link if known, else amazon search)",
    "value_score": 7,
    "original_price": "$399.99 or null if no discount",
    "availability": "In Stock or Limited Stock or Out of Stock"
  }
]

Rules:
- NO hallucinated links to specific product pages unless you are certain they exist
- Use Amazon/Walmart/Flipkart search URLs as fallback links
- value_score must be integer 1-10 (10 = best value)
- Return ONLY the JSON array, nothing else`;

  const response = await fetch("https://api.anthropic.com/v1/messages", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({
      model: MODEL,
      max_tokens: 2000,
      messages: [{ role: "user", content: prompt }],
    }),
  });

  if (!response.ok) throw new Error(`API error: ${response.status}`);
  const data = await response.json();
  const text = data.content?.map(b => b.text || "").join("") || "";
  const clean = text.replace(/```json|```/g, "").trim();
  const match = clean.match(/\[[\s\S]*\]/);
  if (!match) throw new Error("No JSON array found in response");
  return JSON.parse(match[0]);
}

async function fetchChatReply(messages, query) {
  const systemPrompt = `You are NexusMind AI, a smart shopping assistant for the NexusMind product discovery platform. 
You help users find the best products, compare deals, understand specs, and make purchase decisions.
The user is currently searching/browsing: "${query || 'various products'}".
Be concise, friendly, and genuinely helpful. Use emojis sparingly. Keep replies under 120 words.`;

  const response = await fetch("https://api.anthropic.com/v1/messages", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({
      model: MODEL,
      max_tokens: 300,
      system: systemPrompt,
      messages: messages.map(m => ({ role: m.role, content: m.content })),
    }),
  });

  if (!response.ok) throw new Error("Chat API error");
  const data = await response.json();
  return data.content?.[0]?.text || "Sorry, I couldn't process that.";
}

const safeImg = (url) => (url && url.startsWith("http") ? url : FALLBACK_IMG);
const safeLink = (link, name) =>
  link && link.startsWith("http")
    ? link
    : `https://www.amazon.com/s?k=${encodeURIComponent(name || "")}`;

const scoreColor = (s) => {
  if (s >= 8) return "#00d084";
  if (s >= 6) return "#f59e0b";
  return "#ef4444";
};

function ChatBot({ currentQuery }) {
  const [open, setOpen] = useState(false);
  const [messages, setMessages] = useState([
    { role: "assistant", content: "👋 Hi! I'm NexusMind AI. Ask me anything about products, deals, or comparisons!" }
  ]);
  const [input, setInput] = useState("");
  const [loading, setLoading] = useState(false);
  const bottomRef = useRef(null);

  useEffect(() => {
    bottomRef.current?.scrollIntoView({ behavior: "smooth" });
  }, [messages, open]);

  const send = async () => {
    const text = input.trim();
    if (!text || loading) return;
    setInput("");
    const newMsgs = [...messages, { role: "user", content: text }];
    setMessages(newMsgs);
    setLoading(true);
    try {
      const apiMsgs = newMsgs.filter(m => m.role !== "assistant" || newMsgs.indexOf(m) > 0);
      const reply = await fetchChatReply(apiMsgs, currentQuery);
      setMessages([...newMsgs, { role: "assistant", content: reply }]);
    } catch {
      setMessages([...newMsgs, { role: "assistant", content: "⚠️ Something went wrong. Please try again." }]);
    }
    setLoading(false);
  };

  return (
    <>
      {/* Toggle button */}
      <button onClick={() => setOpen(o => !o)} style={{
        position: "fixed", bottom: 28, right: 28, zIndex: 1000,
        width: 58, height: 58, borderRadius: "50%",
        background: "linear-gradient(135deg, #e94560, #c62a47)",
        border: "none", cursor: "pointer", fontSize: 24,
        boxShadow: "0 8px 30px rgba(233,69,96,0.45)",
        display: "flex", alignItems: "center", justifyContent: "center",
        transition: "transform 0.2s",
      }}
        onMouseEnter={e => e.currentTarget.style.transform = "scale(1.1)"}
        onMouseLeave={e => e.currentTarget.style.transform = "scale(1)"}
      >
        {open ? "✕" : "🤖"}
      </button>

      {/* Chat window */}
      {open && (
        <div style={{
          position: "fixed", bottom: 100, right: 28, zIndex: 999,
          width: 340, background: "#0f0f23",
          border: "1px solid rgba(233,69,96,0.3)",
          borderRadius: 20, overflow: "hidden",
          boxShadow: "0 30px 80px rgba(0,0,0,0.5)",
          display: "flex", flexDirection: "column",
          animation: "fadeSlideUp 0.3s ease",
        }}>
          {/* Header */}
          <div style={{
            background: "linear-gradient(135deg, #e94560, #c62a47)",
            padding: "14px 18px", display: "flex", alignItems: "center", gap: 10
          }}>
            <div style={{ fontSize: 22 }}>🤖</div>
            <div>
              <div style={{ color: "#fff", fontWeight: 700, fontSize: 15 }}>NexusMind AI</div>
              <div style={{ color: "rgba(255,255,255,0.7)", fontSize: 11 }}>
                <span style={{ display: "inline-block", width: 7, height: 7, background: "#00d084", borderRadius: "50%", marginRight: 5 }} />
                Online · Smart Shopping Assistant
              </div>
            </div>
          </div>

          {/* Messages */}
          <div style={{ flex: 1, overflowY: "auto", padding: 14, display: "flex", flexDirection: "column", gap: 10, maxHeight: 320 }}>
            {messages.map((m, i) => (
              <div key={i} style={{ display: "flex", justifyContent: m.role === "user" ? "flex-end" : "flex-start" }}>
                <div style={{
                  background: m.role === "user"
                    ? "linear-gradient(135deg, #e94560, #c62a47)"
                    : "rgba(255,255,255,0.06)",
                  color: m.role === "user" ? "#fff" : "#ccd6f6",
                  padding: "9px 13px", borderRadius: 12,
                  fontSize: 13, lineHeight: 1.5, maxWidth: "82%",
                  borderBottomRightRadius: m.role === "user" ? 4 : 12,
                  borderBottomLeftRadius: m.role === "assistant" ? 4 : 12,
                }}>
                  {m.content}
                </div>
              </div>
            ))}
            {loading && (
              <div style={{ display: "flex", gap: 5, padding: "8px 12px", background: "rgba(255,255,255,0.06)", borderRadius: 12, width: "fit-content" }}>
                {[0, 1, 2].map(i => (
                  <div key={i} style={{
                    width: 7, height: 7, borderRadius: "50%", background: "#e94560",
                    animation: `typingBounce 1s ease ${i * 0.15}s infinite`
                  }} />
                ))}
              </div>
            )}
            <div ref={bottomRef} />
          </div>

          {/* Input */}
          <div style={{ padding: "10px 14px", borderTop: "1px solid rgba(255,255,255,0.06)", display: "flex", gap: 8 }}>
            <input
              value={input}
              onChange={e => setInput(e.target.value)}
              onKeyDown={e => e.key === "Enter" && send()}
              placeholder="Ask about any product…"
              style={{
                flex: 1, background: "rgba(255,255,255,0.06)",
                border: "1px solid rgba(255,255,255,0.1)", borderRadius: 8,
                padding: "9px 12px", color: "#ccd6f6", fontSize: 13,
                outline: "none", fontFamily: "inherit"
              }}
            />
            <button onClick={send} disabled={loading} style={{
              background: "#e94560", border: "none", borderRadius: 8,
              padding: "9px 14px", color: "#fff", cursor: "pointer", fontSize: 16,
              opacity: loading ? 0.5 : 1
            }}>➤</button>
          </div>
        </div>
      )}
    </>
  );
}

    const updated = [q, ...recentSearches.filter(r => r !== q)].slice(0, 5);

      const results = await fetchProductsFromAI(q);

      setStatus("success");

      setError(e.message || "Failed to fetch products");
      setStatus("error");

        )}

        {/* SUCCESS */}
        {status === "success" && (
          <>
            {/* Results header + filter */}
            <div style={{
              maxWidth: 1200, margin: "0 auto", padding: "0 5vw 20px",
              display: "flex", alignItems: "center", justifyContent: "space-between",
              flexWrap: "wrap", gap: 12,
              animation: "fadeSlideUp 0.4s ease"
            }}>
              <div>
                <span style={{ color: "#ccd6f6", fontWeight: 700 }}>{products.length} results</span>
                <span style={{ color: "#8892b0" }}> for "</span>
                <span style={{ color: "#e94560" }}>{lastQuery}</span>
                <span style={{ color: "#8892b0" }}>"</span>
              </div>
              <div style={{ display: "flex", gap: 8, alignItems: "center" }}>
                <span style={{ color: "#8892b0", fontSize: 13 }}>Sort:</span>
                {[["value", "Best Value"], ["price_asc", "Price ↑"], ["price_desc", "Price ↓"]].map(([val, label]) => (
                  <button key={val} onClick={() => setSortBy(val)} style={{
                    background: sortBy === val ? "rgba(233,69,96,0.2)" : "rgba(255,255,255,0.04)",
                    border: `1px solid ${sortBy === val ? "rgba(233,69,96,0.6)" : "rgba(255,255,255,0.1)"}`,
                    color: sortBy === val ? "#e94560" : "#8892b0",
                    borderRadius: 8, padding: "6px 14px", fontSize: 12,
                    cursor: "pointer", fontFamily: "inherit", fontWeight: sortBy === val ? 700 : 400,
                  }}>{label}</button>
                ))}
              </div>
            </div>

            {/* Product grid */}
            <div style={s.grid}>
              {sortedProducts.map((product, i) => (
                <ProductCard key={i} product={product} idx={i} />
              ))}
            </div>

            {/* Disclaimer */}
            <p style={{ textAlign: "center", color: "#2d3748", fontSize: 12, marginTop: 32 }}>
              ⚡ Results generated by AI · Prices may vary · Always verify on retailer site · Value scores are AI estimates
            </p>
          </>
        )}
      </main>

      {/* CHATBOT */}
      <ChatBot currentQuery={lastQuery} />
    </div>
  );
}
